16  package com.linecorp.armeria.client.endpoint.zookeeper;
17  
18  import java.nio.charset.StandardCharsets;
19  import java.util.ArrayList;
20  import java.util.List;
21  import java.util.regex.Pattern;
22  
23  import com.linecorp.armeria.client.Endpoint;
24  import com.linecorp.armeria.client.endpoint.EndpointGroupException;
25  
26  /**
27   * Default {@link ZooKeeperNodeValueConverter} implementation which assumes zNode value is a comma-separated
28   * string. Each element of the zNode value represents an endpoint whose format is
29   * {@code <host>[:<port_number>[:weight]]}, such as:
30   * <ul>
31   *   <li>{@code "foo.com"} - default port number, default weight (1000)</li>
32   *   <li>{@code "bar.com:8080} - port number 8080, default weight (1000)</li>
33   *   <li>{@code "10.0.2.15:0:500} - default port number, weight 500</li>
34   *   <li>{@code "192.168.1.2:8443:700} - port number 8443, weight 700</li>
35   * </ul>
36   * Note that the port number must be specified to specify a weight.
37   */
38  public class DefaultZooKeeperNodeValueConverter implements ZooKeeperNodeValueConverter {
39  
40      private static final Pattern SEGMENT_DELIMITER = Pattern.compile("\\s*,\\s*");
41  
42      @Override
43      public List<Endpoint> convert(byte[] data) {
44          List<Endpoint> endpointsList = new ArrayList<>();
45          String valueString = new String(data, StandardCharsets.UTF_8);
46          try {
47              for (String segment : SEGMENT_DELIMITER.split(valueString)) {
48                  final String[] tokens = segment.split(":");
49                  final Endpoint endpoint;
50                  switch (tokens.length) {
51                      case 1: //host
52                          endpoint = Endpoint.of(segment);
53                          break;
54                      case 2: { //host and port
55                          final String host = tokens[0];
56                          final int port = Integer.parseInt(tokens[1]);
57                          if (port == 0) {
58                              endpoint = Endpoint.of(host);
59                          } else {
60                              endpoint = Endpoint.of(host, port);
61                          }
62                          break;
63                      }
64                      case 3: { //host , port , weight
65                          final String host = tokens[0];
66                          final int port = Integer.parseInt(tokens[1]);
67                          final int weight = Integer.parseInt(tokens[2]);
68                          if (port == 0) {
69                              endpoint = Endpoint.of(host).withWeight(weight);
70                          } else {
71                              endpoint = Endpoint.of(host, port, weight);
72                          }
73                          break;
74                      }
75                      default: //unknown
76                          throw new EndpointGroupException(
77                                  "invalid endpoint list: " + segment + '/' + valueString);
78                  }
79  
80                  endpointsList.add(endpoint);
81              }
82          } catch (EndpointGroupException e) {
83              throw e;
84          } catch (Exception e) {
85              throw new EndpointGroupException("invalid endpoint list: " + valueString, e);
86          }
87  
88          if (endpointsList.isEmpty()) {
89              throw new EndpointGroupException("ZNode does not contain any endpoints.");
90          }
91  
92          return endpointsList;
93      }
94  }
